Code prêt à l'emploi

Ce code gère un paiement MTN ou Orange Money via CinetPay :

JavaScript — paiement.js
// NutriCam — Paiement Mobile Money via CinetPay
// Georges Duplece Pitchou | pitchouduplece9@gmail.com

// ── INITIER UN PAIEMENT ──
async function initierPaiement(utilisateur, plan) {
  const montant = plan === 'premium' ? 2500 : 5000;
  
  const reponse = await fetch('https://api-checkout.cinetpay.com/v2/payment', {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({
      apikey: process.env.CINETPAY_API_KEY,
      site_id: process.env.CINETPAY_SITE_ID,
      transaction_id: `TXN-${Date.now()}`,
      amount: montant,
      currency: 'XAF',           // FCFA
      description: `NutriCam ${plan} - Abonnement mensuel`,
      customer_name: utilisateur.nom,
      customer_surname: utilisateur.prenom,
      customer_email: utilisateur.email,
      customer_phone_number: utilisateur.telephone,
      customer_address: 'Yaoundé, Cameroun',
      customer_city: 'Yaoundé',
      customer_country: 'CM',
      customer_state: 'CM',
      customer_zip_code: '00000',
      notify_url: 'https://nutricam.vercel.app/api/webhook-paiement',
      return_url: 'https://nutricam.vercel.app/dashboard',
      channels: 'MOBILE_MONEY',
      metadata: `${utilisateur.id}|${plan}`,
    })
  });
  
  const data = await reponse.json();
  // Redirigez l'utilisateur vers data.data.payment_url
  return data.data.payment_url;
}

// ── RECEVOIR LA CONFIRMATION (Webhook) ──
// URL : /api/webhook-paiement
async function webhookPaiement(requete, reponse) {
  const { cpm_trans_id, cpm_amount, cpm_payment_status, metadata } = requete.body;
  
  if (cpm_payment_status === 'ACCEPTED') {
    const [userId, plan] = metadata.split('|');
    
    // ✅ Activer l'abonnement en base de données
    await activerAbonnement(userId, plan);
    
    // ✅ Envoyer email de confirmation
    await envoyerEmailConfirmation(userId, montant);
    
    console.log(`✅ Paiement OK: ${cpm_amount} FCFA - User: ${userId}`);
  }
  reponse.json({ status: 'reçu' });
}

1
Créer la base de données Supabase (gratuit)
10 min

Allez sur supabase.com → "Start your project" → Connectez avec GitHub

Créez un nouveau projet : "nutricam" → Region : "EU West" (plus proche de vous)

Dans "Table Editor", créez ces 3 tables en cliquant sur "New Table" :

Tables SQL à créer
-- Table 1 : Utilisateurs
CREATE TABLE utilisateurs (
  id UUID DEFAULT uuid_generate_v4() PRIMARY KEY,
  nom TEXT,
  prenom TEXT,
  email TEXT UNIQUE NOT NULL,
  telephone TEXT,
  plan TEXT DEFAULT 'gratuit',
  calories_objectif INTEGER DEFAULT 2000,
  poids DECIMAL,
  cree_le TIMESTAMP DEFAULT now()
);

-- Table 2 : Journal alimentaire
CREATE TABLE journal_repas (
  id UUID DEFAULT uuid_generate_v4() PRIMARY KEY,
  user_id UUID REFERENCES utilisateurs(id),
  recette_nom TEXT,
  calories INTEGER,
  proteines DECIMAL,
  glucides DECIMAL,
  lipides DECIMAL,
  date_repas DATE DEFAULT CURRENT_DATE,
  heure_repas TIME
);

-- Table 3 : Abonnements & Paiements
CREATE TABLE abonnements (
  id UUID DEFAULT uuid_generate_v4() PRIMARY KEY,
  user_id UUID REFERENCES utilisateurs(id),
  plan TEXT,
  montant INTEGER,
  devise TEXT DEFAULT 'XAF',
  methode_paiement TEXT,
  transaction_id TEXT,
  debut_abonnement DATE,
  fin_abonnement DATE,
  statut TEXT DEFAULT 'actif'
);
✅
Une fois les tables créées, Supabase génère automatiquement une API REST pour chaque table. Vous pouvez déjà lire et écrire des données !

Code backend Node.js — Serveur Express complet
Code prêt à l'emploi
server.js — Backend NutriCam complet
// NutriCam Backend — server.js
// Georges Duplece Pitchou | pitchouduplece9@gmail.com
// Démarrage : node server.js

const express = require('express');
const cors = require('cors');
const { createClient } = require('@supabase/supabase-js');
const Anthropic = require('@anthropic-ai/sdk');
require('dotenv').config();

const app = express();
app.use(cors()); app.use(express.json());

// ── CONNEXIONS ──
const supabase = createClient(
  process.env.SUPABASE_URL,
  process.env.SUPABASE_ANON_KEY
);
const ai = new Anthropic({ apiKey: process.env.ANTHROPIC_API_KEY });

// ── ROUTE TEST ──
app.get('/', (req, res) => {
  res.json({ message: '🌿 NutriCam API OK', version: '1.0.0' });
});

// ── INSCRIPTION UTILISATEUR ──
app.post('/api/inscription', async (req, res) => {
  const { nom, prenom, email, telephone } = req.body;
  const { data, error } = await supabase
    .from('utilisateurs')
    .insert([{ nom, prenom, email, telephone }])
    .select();
  if (error) return res.status(400).json({ erreur: error.message });
  res.json({ utilisateur: data[0], message: '✅ Bienvenue sur NutriCam !' });
});

// ── ENREGISTRER UN REPAS ──
app.post('/api/repas', async (req, res) => {
  const { user_id, recette_nom, calories, proteines, glucides, lipides } = req.body;
  const { data, error } = await supabase
    .from('journal_repas')
    .insert([{ user_id, recette_nom, calories, proteines, glucides, lipides }])
    .select();
  if (error) return res.status(400).json({ erreur: error.message });
  res.json({ repas: data[0] });
});

// ── COACH IA NUTRITIONNEL ──
app.post('/api/coach', async (req, res) => {
  const { question, profil_utilisateur } = req.body;
  const reponse = await ai.messages.create({
    model: 'claude-sonnet-4-5', max_tokens: 500,
    system: `Tu es le Coach NutriCam, expert en nutrition africaine et gastronomie camerounaise. Réponds en français, de façon chaleureuse et pratique. Profil: ${JSON.stringify(profil_utilisateur)}`,
    messages: [{ role: 'user', content: question }]
  });
  res.json({ reponse: reponse.content[0].text });
});

// ── WEBHOOK PAIEMENT CINETPAY ──
app.post('/api/webhook-paiement', async (req, res) => {
  const { cpm_payment_status, metadata } = req.body;
  if (cpm_payment_status === 'ACCEPTED') {
    const [userId, plan] = metadata.split('|');
    const fin = new Date(); fin.setMonth(fin.getMonth() + 1);
    await supabase.from('utilisateurs').update({ plan }).eq('id', userId);
    await supabase.from('abonnements').insert([{ user_id: userId, plan, fin_abonnement: fin }]);
  }
  res.json({ status: 'ok' });
});

app.listen(3001, () => console.log('🌿 NutriCam API sur http://localhost:3001'));
